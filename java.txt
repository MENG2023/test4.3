  int[] array = {10,5,3,7,6};
  myBubblesort(array);
  System.out.println(Array.toString(array));
 }
public static void myBubblesort (int[] array) {
 for(int i = 0;i<array.length;i++) {
      for(int j = 0;j<array.length-1;j++) {
        if(array[j] > array[j+1] {
          int tmp = 0;
           tmp = array[j];
           array[j] = array[j+1];
           array[j+1] = tmp;
   }
}
